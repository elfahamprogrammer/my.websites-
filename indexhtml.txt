        <source src="hover.mp3" type="audio/mpeg">
    </audio>

    <script>
        const btn = document.getElementById('followBtn');
        const music = document.getElementById('bgMusic');
        const clickSound = document.getElementById('clickSound');
        const hoverSound = document.getElementById('hoverSound');
        const bio = document.getElementById('bio');
        const profileImg = document.getElementById('profileImg');
        const container = document.querySelector('.container');

        const bioTexts = [
            "مهووس بالجيمينج، عاشق للبرمجة!",
            "بستمتع بكل جديد في التكنولوجيا!",
            "أحب التحديات وأتعلم بسرعة!",
            "صاحب فضول لا ينتهي!"
        ];

        let bioIndex = 0;

        function changeBio() {
            bioIndex = (bioIndex + 1) % bioTexts.length;
            bio.innerHTML = bioTexts[bioIndex];
        }

        setInterval(changeBio, 4000);

        function createParticles(x, y) {
            for (let i = 0; i < 15; i++) {
                const particle = document.createElement('div');
                particle.classList.add('particle');
                const angle = Math.random() * 2 * Math.PI;
                const radius = Math.random() * 80;
                particle.style.left = x + 'px';
                particle.style.top = y + 'px';
                particle.style.setProperty('--x', `${Math.cos(angle) * radius}px`);
                particle.style.setProperty('--y', `${Math.sin(angle) * radius}px`);
                container.appendChild(particle);
                setTimeout(() => particle.remove(), 800);
            }
        }

        btn.addEventListener('mouseenter', () => {
            music.volume = 0.3;
            hoverSound.play();
        });

        btn.addEventListener('mouseleave', () => {
            music.volume = 0.7;
        });

        btn.addEventListener('click', (e) => {
            clickSound.play();
            const rect = btn.getBoundingClientRect();
            createParticles(e.clientX - rect.left, e.clientY - rect.top);
            alert("شكراً لمتابعتك يا محمد! 🎉");
        });

        profileImg.addEventListener('click', (e) => {
            clickSound.play();
            profileImg.style.transform = 'rotate(360deg) scale(1.3)';
            const rect = profileImg.getBoundingClientRect();
            createParticles(rect.width/2, rect.height/2);
            setTimeout(() => {
                profileImg.style.transform = 'rotate(0deg) scale(1)';
            }, 800);
        });
    </script>
